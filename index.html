Sub StartGame() Dim playerChoice As String ' 遊戲介紹 MsgBox "歡迎來到文字冒險遊戲！你正在進入一片神秘的森林，前方有兩條道路。" ' 第一階段：選擇道路 playerChoice = InputBox("你站在分岔路口。你要選擇走哪條路？" & vbCrLf & "輸入 '左' 或 '右'。", "選擇路徑") If playerChoice = "左" Then LeftPath ElseIf playerChoice = "右" Then RightPath Else MsgBox "無效的選擇！遊戲結束。" End If End Sub Sub LeftPath() Dim playerChoice As String MsgBox "你選擇了左邊的道路，走了幾分鐘後，你遇到了一位神秘的老人。" ' 第二階段：與老人互動 playerChoice = InputBox("老人問你：'你是想要財富還是智慧？' " & vbCrLf & "輸入 '財富' 或 '智慧'。", "選擇你的目標") If playerChoice = "財富" Then MsgBox "老人給了你一箱金子，恭喜你獲得了財富！" ElseIf playerChoice = "智慧" Then MsgBox "老人傳授給你一種古老的智慧，你的智慧大增！" Else MsgBox "無效的選擇！你迷失了方向，遊戲結束。" End If End Sub Sub RightPath() Dim playerChoice As String MsgBox "你選擇了右邊的道路，走了一會兒，你發現自己陷入了一個陷阱。" ' 第二階段：逃脫陷阱 playerChoice = InputBox("你想用力跳出陷阱，還是靜下心來觀察？" & vbCrLf & "輸入 '跳' 或 '觀察'。", "逃脫陷阱") If playerChoice = "跳" Then MsgBox "你奮力一跳，成功逃脫了陷阱，獲得了自由！" ElseIf playerChoice = "觀察" Then MsgBox "你仔細觀察到陷阱的弱點，成功逃脫了困境！" Else MsgBox "無效的選擇！你無法逃脫，遊戲結束。" End If End Sub
